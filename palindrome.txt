DELIMITER $$

CREATE PROCEDURE is_palindrome(IN n INT)
BEGIN
    DECLARE original INT DEFAULT n;
    DECLARE rev INT DEFAULT 0;
    DECLARE rem INT;

    WHILE n > 0 DO
        SET rem = MOD(n, 10);
        SET rev = rev * 10 + rem;
        SET n = FLOOR(n / 10);
    END WHILE;

    IF original = rev THEN
        SELECT 'Palindrome';
    ELSE
        SELECT 'Not Palindrome';
    END IF;
END$$

DELIMITER ;
