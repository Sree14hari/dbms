DELIMITER $$

CREATE PROCEDURE is_armstrong(IN n INT)
BEGIN
    DECLARE original INT DEFAULT n;
    DECLARE sum INT DEFAULT 0;
    DECLARE rem INT;

    WHILE n > 0 DO
        SET rem = MOD(n, 10);
        SET sum = sum + POW(rem, 3);
        SET n = FLOOR(n / 10);
    END WHILE;

    IF original = sum THEN
        SELECT 'Armstrong';
    ELSE
        SELECT 'Not Armstrong';
    END IF;
END$$

DELIMITER ;
