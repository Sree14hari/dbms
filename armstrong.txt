CREATE OR REPLACE PROCEDURE armstrong(n IN NUMBER) AS
    original NUMBER := n;
    sum NUMBER := 0;
    rem NUMBER;
BEGIN
    WHILE n > 0 LOOP
        rem := MOD(n, 10);
        sum := sum + rem * rem * rem;
        n := TRUNC(n / 10);
    END LOOP;

    IF sum = original THEN
        DBMS_OUTPUT.PUT_LINE('Armstrong');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Not Armstrong');
    END IF;
END;
/
