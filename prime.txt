CREATE OR REPLACE PROCEDURE is_prime(n IN NUMBER) AS
    i NUMBER := 2;
    flag BOOLEAN := TRUE;
BEGIN
    IF n <= 1 THEN
        flag := FALSE;
    ELSE
        WHILE i <= n / 2 LOOP
            IF MOD(n, i) = 0 THEN
                flag := FALSE;
                EXIT;
            END IF;
            i := i + 1;
        END LOOP;
    END IF;

    IF flag THEN
        DBMS_OUTPUT.PUT_LINE('Prime');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Not Prime');
    END IF;
END;
/
