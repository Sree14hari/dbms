CREATE OR REPLACE PROCEDURE sum_odd_digits(n IN NUMBER) AS
    sum NUMBER := 0;
    rem NUMBER;
BEGIN
    WHILE n > 0 LOOP
        rem := MOD(n, 10);
        IF MOD(rem, 2) = 1 THEN
            sum := sum + rem;
        END IF;
        n := TRUNC(n / 10);
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Sum of odd digits = ' || sum);
END;
/
